* this객체
  - function 키워드
    this: 자신이 소속되어 있는 객체
  ```js
  function BlackDog(){
      this.name = '흰둥이';
      return {
          name:'검둥이',
          bark:function(){
              console.log(this.name+" 멍멍!");
          }
      }
  }
  const bDog = new WhiteDog();
  bDog.bark();
  ```
  - 화살표 함수
    this: 자신이 소속된 외부객체의 인스턴스
  ```js
  function WhiteDog(){
      this.name = '흰둥이';
      return {
          name:'검둥이',
          bark:()=>{
              console.log(this.name+" 멍멍!");
          }
      }
  }
  const wDog = new WhiteDog();
  wDog.bark();
  ```
